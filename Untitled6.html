class CustomBiasLayer(layers.Layer):
    def __init__(self, units, alpha=1.0, **kwargs):
        super().__init__(**kwargs)
        self.units = units  # 
        self.alpha = alpha  # 

    def build(self, input_shape):
        # Wbb_mul
        self.W = self.add_weight(
            shape=(input_shape[-1], self.units),
            initializer="glorot_uniform",
            trainable=True,
            name="kernel"
        )
        self.b = self.add_weight(
            shape=(self.units,),
            initializer="zeros",
            trainable=True,
            name="bias_add"
        )
        self.b_mul = self.add_weight(
            shape=(self.units,),
            initializer="ones",
            trainable=True,
            name="bias_mul"
        )
        super().build(input_shape)

    def call(self, inputs):
        # h = ([xW + b]  b_mul)
        linear = tf.matmul(inputs, self.W) + self.b  # xW + b
        multiplied = linear * self.b_mul  # 
        output = self.alpha * multiplied  # 
        return tf.nn.relu(output)  # ReLU

    def get_config(self):
        # 
        config = super().get_config()
        config.update({"units": self.units, "alpha": self.alpha})
        return config

# 2. Model C
class ModelC(keras.Model):
    def __init__(self, **kwargs):
        super().__init__(**kwargs)
        # 132
        self.custom_layer1 = CustomBiasLayer(32, alpha=0.8, name="custom_layer_1")
        # 216
        self.custom_layer2 = CustomBiasLayer(16, alpha=0.8, name="custom_layer_2")
        # 
        self.output_layer = layers.Dense(3, activation="softmax", name="output_layer")

    def build(self, input_shape):
        self.built = True 

    def call(self, inputs):
        # 
        x = self.custom_layer1(inputs)
        x = self.custom_layer2(x)
        output = self.output_layer(x)
        return output

def evaluate_model(model, x_input, y_true, model_name):
    loss, accuracy = model.evaluate(x_input, y_true, verbose=0)
    print(f"\n{model_name} ")
    print(f": {loss:.4f}")
    print(f": {accuracy:.4f}")
    return loss, accuracy

# 
loss_a, acc_a = evaluate_model(model_a, X_test_scaled, y_test_onehot, "Model A")

def plot_history(history, model_name, subplot_idx):
    # 
    plt.subplot(2, 3, subplot_idx)
    plt.plot(history.history['accuracy'], label='')
    plt.plot(history.history['val_accuracy'], label='')
    plt.title(f'{model_name} ')
    plt.xlabel('Epoch')
    plt.ylabel('')
    plt.legend()
    plt.grid(True, alpha=0.3)

    # 
    plt.subplot(2, 3, subplot_idx + 3)
    plt.plot(history.history['loss'], label='')
    plt.plot(history.history['val_loss'], label='')
    plt.title(f'{model_name} ')
    plt.xlabel('Epoch')
    plt.ylabel('')
    plt.legend()
    plt.grid(True, alpha=0.3)

# 
plot_history(history_a, "Model A", 1)
plot_history(history_b, "Model B", 2)
